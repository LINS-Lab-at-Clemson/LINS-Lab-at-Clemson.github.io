venue = "Journal of Hydrology, 624, 129959"

abstract = """This study develops a coupled agent-based tool for comparing agricultural sediment-management incentives across the Susquehanna River Basin. The model represents farmers' adoption of best management practices, including responses to social norms and performance bonuses, and compares conventional pay-for-practice support with individual- and group-based pay-for-performance programs. Simulations indicate that performance-based policies can reduce more suspended sediment at lower cost, while the group-based design delivers the greatest basin-wide reduction with the least uncertainty and lower sensitivity to bonus levels."""
